import { useState } from "react";

function App(){
  const [skills,set] = useState(["cpp","java","python","js","rust"]);
  function change(ans,k){
    if(k==true){
      set([...skills.filter((item)=>item != ans)])
    }else{
      set([...skills,ans]);
    }
  }
  return(
    <div>
    <h1>Choose Your Skills</h1>
    <input type="checkbox" id="cpp" value="cpp" onChange={(evt)=>change(evt.target.value,evt.target.checked)}/>
    <label htmlFor="cpp">cpp</label>
    <br></br>
    <br></br>
    <input type="checkbox" id="java" value="java" onChange={(evt)=>change(evt.target.value,evt.target.checked)}/>
    <label htmlFor="java">java</label>
    <br></br>
    <br></br>
    <input type="checkbox" id="python" value="python" onChange={(evt)=>change(evt.target.value,evt.target.checked)}/>
    <label htmlFor="python">python</label>
    <br></br>
    <br></br>
    <input type="checkbox" id="js" value="js" onChange={(evt)=>change(evt.target.value,evt.target.checked)}/>
    <label htmlFor="js">JS</label>
    <br></br>
    <br></br>
    <input type="checkbox" id="rust" value="rust" onChange={(evt)=>change(evt.target.value,evt.target.checked)}/>
    <label htmlFor="rust">rust</label>
    <h1>{skills.toString()}</h1>
    </div>
  )
}
export default App;

import { useState } from "react";

function App(){
   const [gender,setgender] = useState("Male");
   const [cou,setcou] = useState("India");

     setcou(ans)
   }
   function change(ans){
     setgender(ans);
   }
  return(
    <div>
      <h1>Select Gender</h1>
      <input type="radio" value="Male" name="gender" id="male" checked={gender=="Male"} 
      onChange={(e)=>change(e.target.value)}/>
      <label htmlFor="male">Male</label>
      <input type="radio" value="Female" name="gender" id="female" checked={gender=="Female"} 
      onChange={(e)=>change(e.target.value)}/>
      <label htmlFor="female">Female</label>
      <input type="radio" value="Transgender" name="gender" id="transgender" checked={gender=="Transgender"} 
      onChange={(e)=>change(e.target.value)}/>
      <label htmlFor="transgender">Transgender</label>
      <h2 style={{marginBottom: "100px"}}>Selected Gender : {gender}</h2>

import { useEffect, useState } from "react";

function App(){
  const [time,set] = useState(0);
  const [col,setcol] = useState("white");
  function change(ans) {
    setcol(ans);
  }
  useEffect(()=>{
      setInterval(()=>{
         set(new Date().toLocaleTimeString());
      },1000);
  },[]);
  return(
    <div>
    <h1>Clock</h1>
    <h2 style={{border:"2px solid black",width:"200px",height:"30px",background: "black"
      ,color:col,textAlign: "center",borderRadius:"15px"}}>{time}</h2>
      <select onChange={(e)=>change(e.target.value)}>
        <option value="white">White</option>
        <option value="red">Red</option>
        <option value="green">Green</option>
        <option value="pink">Pink</option>
        <option value="aqua">Aqua</option>
        <option value=" #cb64ffaa"> #cb64ffaa</option>
      </select>
    </div>
  )
}
export default App;
